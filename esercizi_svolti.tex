\question \textbf{The One-Period Binomial Model} \\
A stock is currently priced at $S_0=100$. In one year, it can either go up to $S_u=120$ or down to $S_d=90$. The risk-free rate is $r=5\%$ (annual compounding).
\begin{parts}
	\part Calculate the risk-neutral probabilities $q$ and $1-q$.
	\part If the "real-world" probability of an up-move is $p=0.7$, calculate the risk premium ($\mu - r$) that investors are demanding in the real world.
\end{parts}

\begin{solution}
	\begin{parts}
		\part In a risk-neutral world, the current price $S_0$ must be the expected future price discounted at the risk-free rate $r$.
		\begin{equation}
			S_0=\frac{1}{1+r}[qS_u +(1-q)S_d]
		\end{equation}
		Plugging in the numbers:
		\begin{gather*}
			100=\frac{1}{1.05}[120\cdot q + 90\cdot(1-q)] \\
			105=120\cdot q+90-90\cdot q
		\end{gather*}
		Hence: $15=30\cdot q \implies q = 0.5$. So, the risk-neutral probabilities are 50\% up and 50\% down.
		\part To find the risk premium ($\mu - r$) we can start from the real-world expected return:
		\begin{equation*}
			\mathbb{E}^{\mathbb{P}}[S_1]=p S_u+(1-p)S_d = 0.7\cdot 120 + 0.3\cdot 90 = 111.0
		\end{equation*}
		This can be compared to just investing into risk-free which brings 105, so the risk premium turns out to be 6\%.		
	\end{parts}
\end{solution}

\question \textbf{Risk-Neutral Measure} \\
Suppose you are in a market with one underlying stock and two different risk-neutral measures, $\mathbb{Q}^1$ and $\mathbb{Q}^2$, both of which exclude arbitrage.
\begin{parts}
	\part Is the market complete? Why or why not?
	\part What does this imply for the pricing of a European Call option on that stock?
	\part How does this relate to the Harrison-Pliska results in your slides?
\end{parts}

\begin{solution}
\begin{parts}
\part No. According to the Second Fundamental Theorem of Asset Pricing, a market is complete if and only if there is a unique risk-neutral measure $\mathbb{Q}$. Since we have two measures ($\mathbb{Q}^1$ and $\mathbb{Q}^2$), the market is incomplete.
\part There is no single "fair" price. In a complete market, you can perfectly replicate the option with a portfolio of the stock and a bond. In an incomplete market, you cannot perfectly replicate it. Therefore, the price is not uniquely determined by the underlying. There exists a range of arbitrage-free prices: any price between $\mathbb{E}^{\mathbb{Q}^1}[payoff]$ and $\mathbb{E}^{\mathbb{Q}^2}[payoff]$ could technically be valid.
\part This highlights that "No Arbitrage" (First Theorem) only ensures that at least one $\mathbb{Q}$ exists. It doesn't guarantee there is only one. Incompleteness usually arises because there are more sources of randomness (e.g., jumps or stochastic volatility) than there are assets to trade.
\end{parts}
\end{solution}

\question \textbf{Martingale Property} \\
Consider the process $Y(t) = 2^{W(t)}$, where $\{W(T):t\geq 0\}$ is a standard Brownian motion. Is this a martingale ?
Show that the exponential SDE
\begin{equation*}
	dX_t = A_t X_tdW_t,\quad X_0=x_0
\end{equation*}
has the following solution
\begin{equation*}
	X_t = x_0 e^{-\frac{1}{2}\int_0^t A_0^2 ds+\int_0^t A_s dB_s}
\end{equation*}

\begin{solution}
To determine if $Y(t)$ is a martingale, we first rewrite the process in base $e$:
\begin{equation*}
	Y(t) = e^{W(t) \ln 2}
\end{equation*}
Let $f(W_t) = e^{\sigma W_t}$ where $\sigma = \ln 2$. We apply Itô's Lemma:
\begin{equation*}
dY_t = \frac{\partial f}{\partial t} dt + \frac{\partial f}{\partial W} dW_t + \frac{1}{2} \frac{\partial^2 f}{\partial W^2} (dW_t)^2
\end{equation*}
The partial derivatives are:
\begin{itemize}
	\item $\frac{\partial f}{\partial t} = 0$
	\item $\frac{\partial f}{\partial W} = \sigma e^{\sigma W_t} = (\ln 2) Y_t$
	\item $\frac{\partial^2 f}{\partial W^2} = \sigma^2 e^{\sigma W_t} = (\ln 2)^2 Y_t$
\end{itemize}
Substituting these into the expression (recalling that $(dW_t)^2 = dt$):
\begin{equation*}
	dY_t = Y_t \left( \frac{(\ln 2)^2}{2} dt + (\ln 2) dW_t \right)
\end{equation*}
For a process to be a martingale, the drift ($dt$) term must be zero. Since $\frac{(\ln 2)^2}{2} \neq 0$, the process \textbf{is not a martingale}. (Specifically, it is a submartingale).
	
\vspace{1em}
	
Define a auxiliary process $Z_t = \ln X_t$. We apply Itô's Lemma to the function $g(X_t) = \ln X_t$:
\begin{equation*}
	dZ_t = \frac{1}{X_t} dX_t - \frac{1}{2 X_t^2} (dX_t)^2
\end{equation*}
From the given SDE, we have $dX_t = A_t X_t dW_t$. Consequently, $(dX_t)^2 = A_t^2 X_t^2 dt$. Substituting these into the expression for $dZ_t$:
\begin{equation*}
	dZ_t = \frac{1}{X_t} (A_t X_t dW_t) - \frac{1}{2 X_t^2} (A_t^2 X_t^2 dt)
\end{equation*}
\begin{equation*}
	dZ_t = A_t dW_t - \frac{1}{2} A_t^2 dt
\end{equation*}
Integrating from $0$ to $t$:
\begin{equation*}
	Z_t - Z_0 = \int_0^t A_s dW_s - \frac{1}{2} \int_0^t A_s^2 ds
\end{equation*}
Since $Z_t = \ln X_t$ and $Z_0 = \ln x_0$:
\begin{equation*}
	\ln \left( \frac{X_t}{x_0} \right) = \int_0^t A_s dW_s - \frac{1}{2} \int_0^t A_s^2 ds
\end{equation*}
Exponentiating both sides yields the final solution:
\begin{equation*}
	X_t = x_0 \exp \left( -\frac{1}{2} \int_0^t A_s^2 ds + \int_0^t A_s dW_s \right)
\end{equation*}
\end{solution}

\question \textbf{The Expectation Paradox} \\
In a stochastic interest rate world, is it true that 
\begin{equation*}
	P(t,T)=\mathbb{E}^{\mathbb{Q}}\left[ \frac{1}{e^{\int_t^T r_s ds}} |\mathcal{F}_t \right] ?
\end{equation*}
\textbf{Hint}: use Jensen's inequality

\begin{solution}
The risk-neutral pricing formula states that the price of a Zero-Coupon Bond (ZCB) is:
\begin{equation}
	P(t,T) = \mathbb{E}^{\mathbb{Q}}\left[ e^{-\int_t^T r_s ds} \middle| \mathcal{F}_t \right]
\end{equation}
This expression is \textbf{true} by the definition of the risk-neutral measure $\mathbb{Q}$, where the money market account $B_t = e^{\int_0^t r_s ds}$ is the numeraire.
	
To explore the paradox, consider the function $f(x) = e^{-x}$. This function is \textcolor{red}{strictly convex} because $f''(x) = e^{-x} > 0$.
	
By \textbf{Jensen's Inequality}, for any convex function $f$ and random variable $X$:
\begin{equation}
	f(\mathbb{E}[X]) \leq \mathbb{E}[f(X)]
\end{equation}
	
Let $X = \int_t^T r_s ds$. Applying the inequality to the bond price formula:
\begin{equation}
	\underbrace{\exp\left( -\mathbb{E}^{\mathbb{Q}}\left[ \int_t^T r_s ds \middle| \mathcal{F}_t \right] \right)}_{\text{Function of Expectation}} < \underbrace{\mathbb{E}^{\mathbb{Q}}\left[ e^{-\int_t^T r_s ds} \middle| \mathcal{F}_t \right]}_{\text{Bond Price } P(t,T)}
\end{equation}
	
The actual bond price $P(t,T)$ is higher than the price calculated by simply using the expected average path of interest rates. 
	
Due to the \textcolor{red}{convexity} of the bond price-yield relationship, volatility in interest rates increases the bond's value. This is why "Convexity Adjustments" are required when moving between different interest rate products (like Eurodollar futures vs. FRAs).

The "paradox" is a psychological and linguistic one rather than a mathematical error. It arises because our intuition often tricks us into thinking that the average price should be determined by the average rate.

Here is why it is called a paradox in a classroom or trading floor setting:
1. The Intuition Trap

If you ask a student: "If the expected average interest rate for the next year is 5%, what is the bond price?"
Most will answer: $e^{-0.05\cdot 1}$.

But as you proved using Jensen's Inequality: $P(t,T)>e^{-\mathbb{E}[r]}$.


The "Paradox" is that the bond is more expensive than the "average" interest rate suggests. Even if your forecast of the interest rate is perfectly correct on average, your bond price calculation will be wrong (too low) if you ignore volatility.
2. The Jensen's Gap

The paradox is essentially the existence of the Convexity Bias.

If rates go down by 1%, the bond price rises by, say, $1.10.

If rates go up by 1%, the bond price falls by only $0.90.

Net result: If rates are volatile but stay at 5% on average, the bond price keeps gaining value from the swings.

3. The Practical Paradox: Futures vs. Forwards

This paradox has a multi-billion dollar consequence in the markets known as the Futures-Forward Basis.

A Forward Rate Agreement (FRA) is priced directly off the expectation.

A Eurodollar Future is "marked-to-market" daily.

Because of the convexity paradox, the Futures rate is always higher than the Forward rate for the same period. If a trader forgets this "paradox" and treats them as equal, they will lose money on every trade because they haven't accounted for the Convexity Adjustment.
In summary, the paradox is:

"To get the right price, you must use the wrong rate (an adjusted one), because the 'correct' expected rate gives you a price that is too low."

Consider a ZCB where the rate $r$ is $5\%$ on average, but volatile ($0\%$ or $10\%$ with $p=0.5$).
\begin{itemize}
	\item \textbf{Naive Price} (using average rate): 
	$e^{-0.05} = 0.9512$
	\item \textbf{Market Price} (average of prices): 
	$0.5(e^{-0.10}) + 0.5(e^{0}) = 0.9524$
\end{itemize}

The bond is \textbf{more expensive} than the average rate implies. 
In markets, this is the \textcolor{red}{Convexity Bias}. 
To price correctly, we must subtract a \textit{convexity adjustment} from the expected future rate.
\end{solution}

\question \textbf{Forward Rate Agreement} \\
Suppose we have a $1\times 4$ FRA with a notional principal of €1 million.  At contract expiration, the 90-day LIBOR at settlement is 6\% and the contract rate is 5.5\%. Calculate the value of the FRA at maturity.
	
\begin{solution}	
Using the standard FRA settlement formula:
\begin{equation*}
	\text{Value} = N \times \frac{(L - K) \times \tau}{1 + L \times \tau}
\end{equation*}
Where $\tau = \frac{90}{360} = 0.25$.
\begin{enumerate}
	\item \textbf{Interest Difference:} $1,000,000 \times (0.06 - 0.055) \times 0.25 = 1,250$
	\item \textbf{Discounting to Settlement:} $\frac{1,250}{1 + (0.06 \times 0.25)} = \frac{1,250}{1.015}$
	\item \textbf{Final Value:} \boxed{€1,231.53}
\end{enumerate}
The long position (the buyer) receives this amount because the market rate ($6\%$) is higher than the contract rate ($5.5\%$).
\end{solution}
	
\question \textbf{Forward Rates} \\	
The 1-year spot rate on US treasury bonds is 9\%, the 2-year spot rate is 9.5\% and the 3-year spot rate is 10\%. 
\begin{parts}
	\part Calculate the implied 1-year ahead, 1-year forward rate, $F(0;1,2)$. Explain why a 1-year forward rate of 9.6\% could not be explained by the market;
	\part calculate the forward rates  $F(0; 2, 3)$ and $F(0; 1,3)$. Is there a link between $F(0;1,2),F(0;2,3)$ and $F(0;1,3)$ ?
\end{parts}
	
\begin{solution}
To solve this exercise, we use the no-arbitrage relationship between spot rates ($s_n$) and forward rates ($F(0;m,n)$). We assume annual discrete compounding for the US Treasury bonds.
\begin{parts}
	\part The relationship between the 1-year spot rate ($s_1$=9\%), the 2-year spot rate ($s_2$=9.5\%), and the forward rate is:
	\begin{equation*}
		(1+s_2)^2=(1+s_1)^1\cdot (1+F(0;1,2))
	\end{equation*}
	
	Solving for $F(0;1,2)$:
	\begin{gather*}
		1+F(0;1,2)=\frac{(1+s_2)^2}{(1+s_1)}=\frac{(1+0.095)^2}{(1+0.09)}=\frac{1.199025}{1.09}\approx 1.100023 \\
		F(0;1,2)\approx 10.0023\%
	\end{gather*}
	
	The implied forward rate of about 10\% is the only rate consistent with the current spot rates that prevents arbitrage.
	If the market forward rate were 9.6\%, it would be "too low" compared to the synthetic forward rate derived from the spot market.
	An arbitrageur could synthesize a loan to "lend" at the higher implied rate (10\%) and "borrow" at the lower market forward rate (9.6\%). Specifically, one could sell a 1-year bond and buy a 2-year bond (creating a synthetic lending at 10\% for the second year) while simultaneously entering a forward contract to borrow at 9.6\%. This would lock in a risk-free profit. In an efficient market, such opportunities are eliminated by trading, forcing the forward rate to equal the implied rate.
	
	\part This is the 1-year forward rate starting 2 years from now. The formula is:
	\begin{gather*}
		(1+s_3)^3=(1+s_2)^2\cdot (1+F(0;2,3)) \\
		1+F(0;2,3)=\frac{(1+s_3)^3}{(1+s_2)^2}=\frac{1.10^3}{1.095^2}=1\frac{1.331}{1.199025}\approx 1.110069 \\
		F(0;2,3) \approx 11.0069\%
	\end{gather*}
	
	This is the annualized 2-year forward rate starting 1 year from now. The formula is:
	\begin{gather*}
		(1+s_3)^3=(1+s_1)^1\cdot (1+F(0;1,3))^2 \\
		1+F(0;1,3)^2=1\frac{1.331}{1.091}\approx 1.221101 \\
		F(0;1,3)=\sqrt{1.221101-1} \approx 1.105034
		F(0;1,3) \approx 10.5034\%
	\end{gather*}
	
	There is a geometric link. The total return of an investment from year 1 to year 3 can be viewed as one 2-year period or two consecutive 1-year periods:
	\begin{equation*}
		(1+F(0;1,3))^2=(1+F(0;1,2))\cdot(1+F(0;2,3))
	\end{equation*}
	
	This shows that the 2-year forward rate is the geometric average of the two consecutive 1-year forward rates.
\end{parts}	
\end{solution}

\question \textbf{Forward Rate Agreement} \\
If the yield curve is upward sloping, will the FRA rate $K$ be higher or lower than the current 6-month EURIBOR ? 

The value of a FRA at time $t$ can be expresses as $V_t=N\tau P(t, T_2)[F(t;T_1, T_2)-K]$. Given that the stochastic bank account $B_t$ is the numearair for the risk-neutral measure $\mathbb{Q}$, explain why it no loget appears explicitly in this valuation formula. What does this "disappearence" imply about the relationship between FRAs and teh replication of interest rate risk ?.

\begin{solution}
In a market characterized by an upward-sloping yield curve, the FRA rate $K$ (which is the forward rate $F(t; T_1, T_2)$ will be higher than the current spot 6-month EURIBOR. This is due to the term structure of interest rates: when the curve is normal (upward-sloping), the market anticipates higher future short-term rates or demands a liquidity premium for longer-dated maturities. Consequently, the discount factor $P(0, T_2)$ decays more rapidly than $P(0, T_1)$, resulting in a forward rate that sits above the current spot rate.

The fact that $B_t$ vanishes from the final expression is not just an algebraic convenience; it carries three fundamental messages in financial engineering:
\begin{itemize}
	\item The Principle of Static Replication: the disappearance of $B_t$ proves that a FRA is a linear instrument that can be perfectly replicated using only traded bonds ($P(t,T_1)$ and $P(t,T_2)$).Since the bank account $B_t$ represents the "path-dependent" cost of rolling over overnight financing, its absence shows that the value of a FRA depends only on the endpoints ($T_1$ and $T_2$) and not on the specific path interest rates take in between.
	\item From Stochastic Expectations to Market Observables: in the starting formula, $B_{T_2}$ is a stochastic variable-we don't know what the bank account value will be in the future. By simplifying the expression, we move from a probabilistic world (expectations of unknown paths) to a market-price world (observed bond prices). This allows a trader to mark-to-market a position using only the current yield curve without needing a complex volatility model (like Vasicek or Hull-White).
	\item Change of Numeraire Intuition: the disappearance of $B_t$ is the "smoking gun" that suggests the existence of the Forward Measure. In the Risk-Neutral measure, we discount by $B_t$, which is stochastic and correlated with the interest rates in the payoff. This correlation makes the expectation hard to compute. The disappearance tells us that there exists a specific "coordinate system" (the T-Forward Measure) where the bond price $P(t,T)$ acts as the discount factor, effectively "absorbing" the stochasticity of the interest rates.
	\item The "No-Convexity" Signal: mathematically, $B_t$ disappears because the FRA payoff is linear. If you were valuing an Interest Rate Caplet (which is a call option on the rate), $B_t$ would not disappear so easily. It would remain "trapped" inside the expectation, requiring a volatility adjustment. The disappearance is the mathematical proof that FRAs have zero convexity bias relative to the forward curve.
\end{itemize}
\end{solution}

\question \textbf{Forward Rate Agreement} \\
Assume that company A has agreed to pay a 6-month Libor and receive a fixed interest rate of 8\% per annum (with interest payable every six months) from the face value of 100 million. Swap is 1.25 years to expire. The interest rates for 3, 9 and 15 months are: 10\%, 10.5\% and 11\% respectively. Assume that interest rates are continuously compounded. The 6-month Libor is currently 10.2\%. Calculate the value of this swap for company A. Determine the value of the swap using the relationship between interest rate swap and FRA.

Which of the following strategies is (are) appropriate ?
\begin{itemize}
	\item If a borrower has a fixed rate debt and is expecting interest rates to rise, then the borrower should not enter into a swap.
	\item If an investor has floating rate assets and is expecting interest rates to rise, then the investor should enter into a receiver swap.
	\item If a borrower has floating rate debt and is expecting interest rates to rise, then the borrower should enter into a payer swap.
	\item If an investor has fixed income assets and is expecting interest rates to drop, then the investor should enter into a payer swap.
\end{itemize}

\begin{solution}
Company A is a Receiver, meaning they receive 8\% Fixed and pay 6-month LIBOR.

The swap expires in 1.25 years (15 months). Since payments are every 6 months, the payment dates relative to today ($t=0$) are: $T_1=0.25$ years (3 months), $T_2=0.75$ years (9 months) and $T_3=1.25$ years (15 months).

Using continuous compounding $P(0,T)=e^{-rT}$: $P(0,0.25)=e^{-0.10\cdot 0.25}\approx 0.9753$, $P(0,0.75)=e^{-0.105\cdot 0.75}\approx 0.9243$ and $P(0,1.25)=e^{-0.11\cdot 1.25}\approx 0.8715$.

The fixed rate is $K=8\%$. Payments are semi-annual, so $\tau=0.5$.
\begin{equation*}
\text{Fixed Leg Value} = K\cdot A = 100\cdot 0.5\cdot 0.08\cdot (0.9753+0.9243+0.8715) = \$11.0844 M
\end{equation*}

The floating leg instead has a first payment $T_1$ this rate was already "fixed" 3.75 months ago (the current LIBOR of 10.2\%) so Flow1$= N\tau 0.102\cdot 0.9753 = \$4.9740 M$.
For the second and third payments we use the implied Forward Rates: $N[P(0,T_1)-P(0,T_3)] = 100\cdot (0.9753-0.8715)=\$10.38 M$.

The total Swap Value to Company A $\text{Fixed Leg}-\text{Floating Leg} = 11.0844-(4.9740+10.38)=11.0844-15.354=-\$4.2696 M$

Which strategies are appropriate?
\begin{itemize}
	\item CORRECT. If rates rise, their fixed debt becomes "cheaper" relative to market rates. Entering a payer swap would convert them to floating, which is exactly what they want to avoid if rates are rising.
	\item INCORRECT. If rates rise, the floating assets will pay more. A receiver swap would convert that into a fixed rate, capping their gains. They should stay floating.
	\item CORRECT. This is a standard hedge. The payer swap pays floating (canceling the debt payment) and receives floating, leaving the borrower with a fixed rate obligation.
	\item INCORRECT. If rates drop, fixed assets gain value. A payer swap would offset this by paying a fixed rate (which is now high) and receiving a floating rate (which is now low).
\end{itemize}
\end{solution}

\question \textbf{Asset Swap} \\
Describe the asset swap contract for a coupon bond with coupons equal to \textbf{C} and dirty Price equal to $P(t)$. What will happen to the price if the spread increases, all the rest being equal ? What does it mean in term of credit quality expectation by the market for the bond's issuer ?

Consider the 10-year German Bund DBR 0.5\% 2026 which is currently trading at a clean price of 104.58. 

Given that the 10-year EUR swap rate is 0.44\% what is the par-par asset swap spread for this bond ? 
For this exercise assume that all annuity factors have a value of 10.0 for simplicity.

Consider the 10-year Greek Government Bond GGB 3.0\% 2026 which is currently trading at a clean price of 75.280. Given that the 10-year EUR swap rate is 0.440\% what is the par-par asset swap spread for this bond ?

\begin{solution}
The Asset Swap (AS) for a coupon bond is a package consisting of:
\begin{itemize}
\item Buying the risky bond at its dirty price $P(t)$.
\item Entering a swap where you pay the bond's fixed coupons C and receive floating LIBOR/EURIBOR plus a spread $s$.
\item An upfront payment (or exchange of notional) such that the net initial outlay for the investor is exactly Par (100).
\end{itemize}

The Asset Swap Spread $s$ is the "price of risk." Using the simplified relationship:
\begin{equation*}
s = \text{AnnuityRisk} -  \text{Free Price} - \text{Market Risky Price}
\end{equation*}

If $s$ increases (all else being equal), the Price $P(t)$ must decrease.
An increasing spread indicates a deterioration in credit quality expectations. The market is demanding a higher premium to hold the issuer's debt because the perceived probability of default has risen or the liquidity of the bond has dried up.

Clean Price = 104.58 (Since we aren't given accrued interest, we assume Dirty$\approx$Clean for the calculation), Fixed Coupon $K_{bond}=0.5\%$, Swap Rate $S_{10y}=0.44\%$ and Annuity $A=10.0$.

The "Risk-Free" price is the price the bond would have if it were priced exactly at the swap curve ($K=S$): $CBP=100+(Coupon-Swap Rate)\cdot A = 100+(0.5\%-0.44\%)\cdot 10=100.60$.

\begin{equation*}	
s=A\cdot (CBP - CBP) = 10\cdot (100.6-104.58) = -0.398\%
\end{equation*}

The Asset Swap Spread is -39.8 bps.
Note: It is normal for Bunds to have a negative spread because they are often considered "safer" than the interbank swap market.

Clean Price = 75.28, Fixed Coupon $K_{bond}=3.0\%$, Swap Rate $S_{10y}=0.44\%$, Annuity $A=10.0$ (assuming the same for comparison).

\question \textbf{DV01} \\
Consider a 2-year Interest Rate Swap on a notional of 1M, with a fixed rate of 5\% and paying Libor rate annually. The term-structure of interest rates is flat at 5\%.
Estimate the DV01 of the swap numerically. 

\begin{solution}
Calculate the initial NPV, since $K$ equals the flat market rate (5\%), the Swap is at par: $NPV_0=0$.
Bump the curve by +1 bp (aka r=5.01\%). We use the Payer Swap perspective (Pay Fixed, Receive Floating).
\begin{align*}
\text{Fixed Leg} &= N K [P(0,1)+P(0,2)]=1M 0.05 [e^{-0.0501\cdot 1} + e^{-0.0501\cdot 2}] \approx \\
&\approx 95112.7 + 90465.4 = 185578.1 \\
\text{Floating Leg} &= N [1-P(0,2)] = 1M [1-e^{-0.0501\cdot 2}] \approx 1M [1-0.904654] = 95346\\
NPV + 1 bp &\approx 95346-185578\approx = -90232
\end{align*}
(Note: this logic assumes the float leg is valued via the $1-P$ shortcut).

For a par swap on a flat curve, the DV01 is approximately the Annuity times the change in rate:
\begin{equation*}
DV01 \approx N\cdot\tau\cdot\sum P(0,T_i)\cdot 0.0001 \approx 1000000\cdot [0.9512+0.9048]\cdot 0.0001 \approx 185.60
\end{equation*}
\end{solution}

\question \textbf{Cap-Floor Parity} \\
Prove that the difference between the price of a Cap and a Floor with the same strike $K$ and the same maturity structure is equal to the value of a Payer Interest Rate Swap (IRS).

\begin{solution}
Consider a single accrual period $[T_{i-1}, T_i]$. The payoffs at maturity $T_i$ for a Caplet and a Floorlet are defined as:
\begin{itemize}
	\item \textbf{Caplet:} $N \tau_i (L(T_{i-1}, T_i) - K)^+$
	\item \textbf{Floorlet:} $N \tau_i (K - L(T_{i-1}, T_i))^+$
\end{itemize}

By subtracting the Floorlet payoff from the Caplet payoff, we obtain:
\begin{equation*}
	N \tau_i \left[ (L(T_{i-1}, T_i) - K)^+ - (K - L(T_{i-1}, T_i))^+ \right]
\end{equation*}

Using the algebraic identity $(x - k)^+ - (k - x)^+ = x - k$, the resulting difference in payoff is:
\begin{equation*}
	N \tau_i (L(T_{i-1}, T_i) - K)
\end{equation*}

This expression is exactly the payoff of one exchange in a \textbf{Payer Swap} (where one pays the fixed rate $K$ and receives the floating rate $L$). 

Summing these payoffs over all periods from $i = \alpha+1$ to $\beta$ and discounting them back to time $t$ under the risk-neutral measure $\mathbb{Q}$, we get:
\begin{equation*}
	\sum_{i=\alpha+1}^{\beta} \text{Cpl}_i(K) - \sum_{i=\alpha+1}^{\beta} \text{Flr}_i(K) = \mathbb{E}^{\mathbb{Q}} \left[ \sum_{i=\alpha+1}^{\beta} e^{-\int_t^{T_i} r_s ds} N \tau_i (L(T_{i-1}, T_i) - K) \middle| \mathcal{F}_t \right]
\end{equation*}

Which simplifies to the fundamental parity relationship:
\begin{equation}
	\text{Cap}(t, K) - \text{Floor}(t, K) = \text{Payer Swap}(t, K)
\end{equation}
\end{solution}
